```typescript
async function tick(): Promise<void> {
  const campaigns = (await db.listCampaigns()).filter((c) => c.status === "running");
  if (campaigns.length === 0) return;

  let firedThisTick = 0;
  for (const campaign of campaigns) {
    if (firedThisTick >= BATCH_PER_TICK) break;
    if (!campaign.accountIds.length) continue;

    // Per-campaign global cooldown.
    const sinceLast = Date.now() - (campaignLastSent.get(campaign.id) || 0);
    const minSpacingMs = (campaign.minInterSendSeconds || 1800) * 1000;
    if (sinceLast < minSpacingMs) continue;

    const pending = await db.pickPendingLeads(campaign.id, BATCH_PER_TICK - firedThisTick);
    if (pending.length === 0) {
      // No more pending. If nothing sent either, mark campaign finished.
      const remaining = await db.pickPendingLeads(campaign.id, 1);
      if (remaining.length === 0) {
        await db.setCampaignStatus(campaign.id, "finished");
        publishExternalEvent({ type: "campaign_finished", campaignId: campaign.id, ts: new Date().toISOString() });
        console.log(`[campaign] ${campaign.id} finished — no more pending leads`);
      }
      continue;
    }

    for (const lead of pending) {
      if (firedThisTick >= BATCH_PER_TICK) break;

      const accountId = pickAccountForCampaign(campaign);
      if (!accountId) {
        console.warn(`[campaign] ${campaign.id} no usable account`);
        continue;
      }
      const allowed = await isUnderRateLimit(accountId, campaign.ratePerHour, campaign.ratePerDay);
      if (!allowed) continue;

      const token = _getCapturedToken(accountId);
      if (!token) continue;

      // The one path. Lead's DM channel MUST already exist (from wave-prep).
      // If it doesn't, this lead stays pending — operator hasn't waved them yet.
      const channelId = await db.findExistingDmChannel(accountId, lead.discordUserId);
      if (!channelId) {
        // Not yet warm. Skip silently; wave queue will handle it.
        continue;
      }

      const body = pickTemplateBody(campaign, lead);
      accountLastSent.set(accountId, Date.now());
      campaignLastSent.set(campaign.id, Date.now());
      firedThisTick++;

      const sendRes = await sendDiscordMessage(accountId, token, channelId, body);
      if (sendRes.ok) {
        await db.setLeadStatus(lead.id, "sent", accountId);
        await db.bumpCampaignTotal(campaign.id, "sent");
        publishExternalEvent({ type: "fr_sent", campaignId: campaign.id, leadId: lead.id, ts: new Date().toISOString() });
        console.log(`[campaign] sent campaign=${campaign.id} account=${accountId} → ${lead.discordUserId}`);
      } else {
        const cls = classifyDiscordError(sendRes.httpStatus, sendRes.error);
        if (cls.fatal) {
          await panicPauseCampaign(campaign, accountId, lead.id, cls.reason);
          return;
        }
        await db.setLeadStatus(lead.id, "failed", accountId, sendRes.error || `HTTP ${sendRes.httpStatus}`);
        await db.bumpCampaignTotal(campaign.id, "failed");
        publishExternalEvent({ type: "fr_declined", campaignId: campaign.id, leadId: lead.id, ts: new Date().toISOString() });
      }
    }
  }
}

function pickTemplateBody(campaign: db.CampaignRow, lead: db.LeadRow): string {
  const variants = (campaign.templates || []).map((t) => t.trim()).filter(Boolean);
  if (variants.length === 0) return "hey 👋";
  const chosen = variants[Math.floor(Math.random() * variants.length)];
  return chosen
    .replace(/\{\{firstName\}\}/g, lead.displayName || "")
    .replace(/\{\{username\}\}/g, lead.displayName || "");
}
```

```typescript
async function isUnderRateLimit(accountId: string, perHour: number, perDay: number): Promise<boolean> {
  const now = Date.now();
  const requestedSpacing = perHour > 0 ? (3600_000 / perHour) : 0;
  const minSpacing = Math.max(requestedSpacing, DM_MIN_SPACING_MS);  // always 4hr floor
  const last = accountLastSent.get(accountId) || 0;
  if (now - last < minSpacing) return false;

  const oneHourAgo = new Date(now - 3600_000).toISOString();
  const oneDayAgo = new Date(now - 86400_000).toISOString();
  const [sentLastHour, sentLastDay] = await Promise.all([
    db.countLeadsSentSince(accountId, oneHourAgo),
    db.countLeadsSentSince(accountId, oneDayAgo),
  ]);
  const dailyCap = Math.min(perDay, DM_MAX_PER_DAY);  // always 6/day floor
  if (sentLastHour >= perHour) return false;
  if (sentLastDay >= dailyCap) return false;
  return true;
}
```
